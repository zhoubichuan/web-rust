# 冒泡排序

```rust
use rand::Rng;
fn main(){
    println!("随机10个数");
    let mut numbers = [0u32;10];
    for i in 0..10 {
        numbers[i] = rand::thread_rng().gen_range(1..10);
    }
    println!("ERRORE:{:?}",numbers);

    let len = numbers.len();

    for i in 0..len {
        for j in 0..len -i -1{
            if numbers[j]>numbers[j+1]{
                numbers.swap(j, j+1)
            }
        }
    }
    println!("AFTER:{:?}",numbers)
}
```
