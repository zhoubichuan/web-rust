#[wasm_bindgen]
pub fn markdown(enc_data: &str) -> String {
    let arena = Arena::new();

    let root = parse_document(&arena, enc_data, &Options::default());

    let mut html = vec![];
    format_html(root, &Options::default(), &mut html).unwrap();
    String::from_utf8(html).unwrap()
}
