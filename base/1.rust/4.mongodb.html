# 数据类型

# 1.整数类型

  • 有符号整数i前缀
let number: i32 = 42
1
  • 无符号整数u前缀
let number: u64 = 100
1

# 2.浮点数类型

  • 单精度浮点数
let float_number: f32 =3.14
1
  • 双精度浮点数
let double_number: f64 = 3.1415926
1

# 3.布尔类型

  • 值为true或false
let value:bool = true
1

# 4.字符类型

  • 字符类型使用单引号'表示
let word:char = 'A'
1

# 5.数组类型

  • 数组固定大小，且所有元素类型必须相同，使用[]表示
let arr:[i32; 3] = [1,2,3]
1

# 6.元祖类型

  • 存储多个不同类型的值，使用()表示
let tuple:(i32,f64,char)=(42,3.14,'A')
1

# 7.引用类型

  • 引用类型用于引用其他变量的值，而不是拥有自己的所有权，使用&符号表示
let val:i32 = 42
let refer: &i32=&value
1
2

# 8.切片类型

  • 引用数组或向量的一部分数据，使用&[T]表示
let arr:[i32;5]=[1,2,3,4,5]
let slice:&[i32]=&arr[...3]
1
2

# 9.字符串类型

  • 存储文本数据，使用String表示
let val :String = String::from('hello')
1

# 10.向量类型

  • 存储可变大小的元素序列，使用Vec<T>表示
let mut vector:Vec<i32> = Vec::new()
vector.push(1)
vector.push(2)
vector.push(3)
