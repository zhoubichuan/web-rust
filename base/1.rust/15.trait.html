# 1.定义trait

# 1.1 trait 基础

trait Circle {
    fn get_width(&self)->i32
    fn get_height(&self) ->i32
}
struct Value {
    width:i32,
    height:i32
}
impl Value {
    fn new(widht:i32,height:i32)->Self{
        Self{
            width,
            height
        }
    }
}
impl Circle for Value {
    fn get_width(&self)->i32{
        self.width
    }
    fn get_height(&self)->i32{
        self.height
    }
}
